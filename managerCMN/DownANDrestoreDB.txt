



🔹 Cách restore file .bak trên Windows 11
✅ Bước 1: Mở SSMS
Mở Microsoft SQL Server Management Studio
Kết nối tới SQL Server (localhost hoặc server của bạn)
✅ Bước 2: Mở chức năng Restore
Chuột phải vào Databases
Chọn Restore Database...
✅ Bước 3: Chọn file .bak
Chọn Device
Nhấn ... → Add
Chọn file:
managerCMN_Prod.bak
✅ Bước 4: Đặt tên database
Ở ô Database → nhập tên mới
👉 Ví dụ:
managerCMN_Prod_Restore
✅ Bước 5: Cấu hình (quan trọng)

Vào tab Options:

✔ Tick:

Overwrite the existing database (WITH REPLACE) (nếu restore đè)

✔ Kiểm tra đường dẫn:

MDF / LDF → nên để mặc định:
C:\Program Files\Microsoft SQL Server\...
✅ Bước 6: Restore
Nhấn OK
Đợi vài giây → Done 🎉

👉 Dán toàn bộ:

```bash
#!/bin/bash

# ====== CONFIG ======
SQL_SERVER="localhost"
SQL_USER="user"
SQL_PASS="password"
DB_NAME="managerCMN_Prod"

REMOTE_BACKUP_DIR="/var/opt/mssql/backup"
LOCAL_BACKUP_DIR="/root/backup_sql_crm"
LOG_DIR="/root/backup_sql_crm/logs"

SERVER_IP="IP Server"
SERVER_USER="root"

SQLCMD_PATH="/opt/mssql-tools18/bin/sqlcmd"

DATE=$(date +%F)
FILE_NAME="${DB_NAME}_${DATE}.bak"
LOG_FILE="$LOG_DIR/backup_$DATE.log"

# ====== CREATE DIR ======
mkdir -p $LOCAL_BACKUP_DIR
mkdir -p $LOG_DIR

echo "==============================" >> $LOG_FILE
echo "TIME: $(date)" >> $LOG_FILE
echo "START BACKUP DATABASE: $DB_NAME" >> $LOG_FILE

# ====== BACKUP DATABASE ON SERVER ======
ssh $SERVER_USER@$SERVER_IP "
$SQLCMD_PATH -S $SQL_SERVER -U $SQL_USER -P '$SQL_PASS' -C -Q \"
BACKUP DATABASE [$DB_NAME]
TO DISK = '$REMOTE_BACKUP_DIR/$FILE_NAME'
WITH FORMAT, NAME = 'Full Backup';
\"

# DELETE OLD FILES > 90 DAYS
find $REMOTE_BACKUP_DIR -type f -name '*.bak' -mtime +90 -delete
"

if [ $? -eq 0 ]; then
    echo "[OK] Backup on server success: $FILE_NAME" >> $LOG_FILE
else
    echo "[ERROR] Backup on server FAILED!" >> $LOG_FILE
    exit 1
fi

# ====== COPY FILE TO LOCAL ======
scp $SERVER_USER@$SERVER_IP:$REMOTE_BACKUP_DIR/$FILE_NAME $LOCAL_BACKUP_DIR/

if [ $? -eq 0 ]; then
    echo "[OK] Copy to local success" >> $LOG_FILE
else
    echo "[ERROR] Copy FAILED!" >> $LOG_FILE
    exit 1
fi

# ====== DELETE OLD FILES LOCAL > 180 DAYS ======
find $LOCAL_BACKUP_DIR -type f -name "*.bak" -mtime +180 -delete
echo "[OK] Clean old local files (>180 days)" >> $LOG_FILE

# ====== CHECK FILE EXISTS ======
if [ -f "$LOCAL_BACKUP_DIR/$FILE_NAME" ]; then
    FILE_SIZE=$(du -h "$LOCAL_BACKUP_DIR/$FILE_NAME" | cut -f1)
    echo "[OK] File exists: $FILE_NAME | Size: $FILE_SIZE" >> $LOG_FILE
else
    echo "[ERROR] File not found after copy!" >> $LOG_FILE
fi

echo "END TIME: $(date)" >> $LOG_FILE
echo "==============================" >> $LOG_FILE
echo "" >> $LOG_FILE
```

# 🔥 SCRIPT RESTORE (FINAL - có tham số)

```bash
nano /root/restore_sql_crm.sh
```

👉 Dán:

```bash
#!/bin/bash

# ====== CONFIG ======
SQL_SERVER="localhost"
SQL_USER="user"
SQL_PASS="password"
DB_NAME="managerCMN_Prod"

BACKUP_DIR="/root/backup_sql_crm"
LOG_DIR="/root/backup_sql_crm/logs"

SERVER_IP="IP server"
SERVER_USER="root"

SQLCMD_PATH="/opt/mssql-tools18/bin/sqlcmd"

DATE=$(date +%F)
LOG_FILE="$LOG_DIR/restore_$DATE.log"

mkdir -p $LOG_DIR

echo "==============================" >> $LOG_FILE
echo "TIME: $(date)" >> $LOG_FILE
echo "START RESTORE DATABASE: $DB_NAME" >> $LOG_FILE

# ====== XỬ LÝ FILE INPUT ======
INPUT_FILE="$1"

if [ -n "$INPUT_FILE" ]; then
    # Nếu có truyền đường dẫn
    if [ -f "$INPUT_FILE" ]; then
        RESTORE_FILE="$INPUT_FILE"
        echo "[INFO] Dùng file chỉ định: $RESTORE_FILE" >> $LOG_FILE
    else
        echo "[ERROR] File không tồn tại: $INPUT_FILE" >> $LOG_FILE
        exit 1
    fi
else
    # Nếu không truyền → lấy file mới nhất
    RESTORE_FILE=$(ls -t $BACKUP_DIR/*.bak 2>/dev/null | head -n 1)

    if [ -z "$RESTORE_FILE" ]; then
        echo "[ERROR] Không tìm thấy file backup!" >> $LOG_FILE
        exit 1
    fi

    echo "[INFO] Dùng file mới nhất: $RESTORE_FILE" >> $LOG_FILE
fi

FILE_NAME=$(basename "$RESTORE_FILE")

# ====== UPLOAD FILE ======
scp "$RESTORE_FILE" $SERVER_USER@$SERVER_IP:/var/opt/mssql/backup/

if [ $? -ne 0 ]; then
    echo "[ERROR] Upload file lên server thất bại!" >> $LOG_FILE
    exit 1
fi

echo "[OK] Upload file thành công" >> $LOG_FILE

# ====== RESTORE DATABASE ======
ssh $SERVER_USER@$SERVER_IP "
$SQLCMD_PATH -S $SQL_SERVER -U $SQL_USER -P '$SQL_PASS' -C -Q \"
ALTER DATABASE [$DB_NAME] SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
RESTORE DATABASE [$DB_NAME]
FROM DISK = '/var/opt/mssql/backup/$FILE_NAME'
WITH REPLACE;
ALTER DATABASE [$DB_NAME] SET MULTI_USER;
\"
"

if [ $? -eq 0 ]; then
    echo "[OK] Restore database thành công!" >> $LOG_FILE
else
    echo "[ERROR] Restore database thất bại!" >> $LOG_FILE
    exit 1
fi

echo "END TIME: $(date)" >> $LOG_FILE
echo "==============================" >> $LOG_FILE
echo "" >> $LOG_FILE
```

```bash
#!/bin/bash

# ====== CONFIG ======
SQL_SERVER="localhost"
SQL_USER="SA"
SQL_PASS="password"
DB_NAME="managerCMN_Prod"

REMOTE_BACKUP_DIR="/var/opt/mssql/backup"
LOCAL_BACKUP_DIR="/root/backup_sql_crm"
LOG_DIR="/root/backup_sql_crm/logs"

SERVER_IP="ip server"
SERVER_USER="root"

SQLCMD_PATH="/opt/mssql-tools18/bin/sqlcmd"

UPLOAD_FILE="uploads_${DATE}.tar.gz"

mkdir -p $LOG_DIR

echo "==============================" >> $LOG_FILE

    echo "[ERROR] Backup server FAILED!" >> $LOG_FILE
    exit 1

# ====== COPY VỀ LOCAL ======
scp $SERVER_USER@$SERVER_IP:$REMOTE_BACKUP_DIR/$DB_FILE $LOCAL_BACKUP_DIR/
scp $SERVER_USER@$SERVER_IP:$REMOTE_BACKUP_DIR/$UPLOAD_FILE $LOCAL_BACKUP_DIR/

if [ $? -ne 0 ]; then
    echo "[ERROR] Copy FAILED!" >> $LOG_FILE
    exit 1
fi

echo "[OK] Copy về local thành công" >> $LOG_FILE

# ====== DELETE OLD FILE LOCAL (>180 ngày) ======
find $LOCAL_BACKUP_DIR -type f -name "*.bak" -mtime +180 -delete
find $LOCAL_BACKUP_DIR -type f -name "uploads_*.tar.gz" -mtime +180 -delete

echo "[OK] Clean local old files" >> $LOG_FILE

# ====== CHECK ======

    SIZE=$(du -h "$LOCAL_BACKUP_DIR/$DB_FILE" | cut -f1)

fi

if [ -f "$LOCAL_BACKUP_DIR/$UPLOAD_FILE" ]; then
    SIZE=$(du -h "$LOCAL_BACKUP_DIR/$UPLOAD_FILE" | cut -f1)
    echo "[OK] Uploads: $UPLOAD_FILE | Size: $SIZE" >> $LOG_FILE
fi

echo "END TIME: $(date)" >> $LOG_FILE
echo "==============================" >> $LOG_FILE
```

# 🔥 2. SCRIPT RESTORE FULL (DB + FILE)

```bash
#!/bin/bash

# ====== CONFIG ======
SQL_SERVER="localhost"
SQL_USER="SA"
SQL_PASS="password"
DB_NAME="managerCMN_Prod"

BACKUP_DIR="/root/backup_sql_crm"
LOG_DIR="/root/backup_sql_crm/logs"

SERVER_IP="ip server"
SERVER_USER="root"

SQLCMD_PATH="/opt/mssql-tools18/bin/sqlcmd"

UPLOAD_DIR="/var/www/cmnmanager-src/managerCMN/managerCMN/managerCMN/bin/Release/publish/wwwroot/uploads"

DATE=$(date +%F)
LOG_FILE="$LOG_DIR/restore_$DATE.log"

mkdir -p $LOG_DIR

echo "==============================" >> $LOG_FILE
echo "TIME: $(date)" >> $LOG_FILE
echo "START RESTORE FULL SYSTEM" >> $LOG_FILE

# ====== FILE INPUT ======
INPUT_FILE="$1"

if [ -n "$INPUT_FILE" ]; then
    if [ -f "$INPUT_FILE" ]; then
        RESTORE_FILE="$INPUT_FILE"
    else
        echo "[ERROR] File không tồn tại!" >> $LOG_FILE
        exit 1
    fi
else
    RESTORE_FILE=$(ls -t $BACKUP_DIR/*.bak 2>/dev/null | head -n 1)
fi

if [ -z "$RESTORE_FILE" ]; then
    echo "[ERROR] Không có file backup!" >> $LOG_FILE
    exit 1
fi

DB_FILE=$(basename "$RESTORE_FILE")

# ====== UPLOAD FILE DB ======
scp "$RESTORE_FILE" $SERVER_USER@$SERVER_IP:/var/opt/mssql/backup/

# ====== UPLOAD FILE UPLOAD ======
UPLOAD_FILE=$(ls -t $BACKUP_DIR/uploads_*.tar.gz | head -n 1)
UPLOAD_NAME=$(basename $UPLOAD_FILE)

scp "$UPLOAD_FILE" $SERVER_USER@$SERVER_IP:/var/opt/mssql/backup/

echo "[OK] Upload file xong" >> $LOG_FILE

# ====== RESTORE TRÊN SERVER ======
ssh $SERVER_USER@$SERVER_IP "

# RESTORE DB
$SQLCMD_PATH -S $SQL_SERVER -U $SQL_USER -P '$SQL_PASS' -C -Q \"
ALTER DATABASE [$DB_NAME] SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
RESTORE DATABASE [$DB_NAME]
FROM DISK = '/var/opt/mssql/backup/$DB_FILE'
WITH REPLACE;
ALTER DATABASE [$DB_NAME] SET MULTI_USER;
\"

# RESTORE UPLOADS
rm -rf $UPLOAD_DIR/*
tar -xzpf /var/opt/mssql/backup/$UPLOAD_NAME -C /

# FIX PERMISSION (QUAN TRỌNG)
chown -R www-data:www-data $UPLOAD_DIR
chmod -R 755 $UPLOAD_DIR
"

if [ $? -eq 0 ]; then
    echo "[OK] Restore FULL thành công" >> $LOG_FILE
else
    echo "[ERROR] Restore FAILED!" >> $LOG_FILE
    exit 1
fi

echo "END TIME: $(date)" >> $LOG_FILE
echo "==============================" >> $LOG_FILE
echo "" >> $LOG_FILE
```
